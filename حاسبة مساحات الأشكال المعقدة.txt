<canvas id="canvas"></canvas>

<script>
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    let points = [];
    const pixelsToMeter = 20; // كل 20 بكسل تعادل 1 متر (افتراضياً)

    function resize() {
        canvas.width = window.innerWidth;
        canvas.height = window.innerHeight;
        draw();
    }

    window.addEventListener('resize', resize);
    resize();

    canvas.addEventListener('click', (e) => {
        points.push({ x: e.clientX, y: e.clientY });
        if (navigator.vibrate) navigator.vibrate(30);
        draw();
        calculateArea();
    });

    function draw() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
        
        if (points.length === 0) return;

        ctx.beginPath();
        ctx.moveTo(points[0].x, points[0].y);
        ctx.strokeStyle = '#2c3e50';
        ctx.lineWidth = 2;

        points.forEach((p, i) => {
            ctx.lineTo(p.x, p.y);
            // رسم نقطة عند كل زاوية
            ctx.fillStyle = '#d4af37';
            ctx.fillRect(p.x-4, p.y-4, 8, 8);
        });

        if (points.length > 2) {
            ctx.closePath();
            ctx.fillStyle = 'rgba(212, 175, 55, 0.2)';
            ctx.fill();
        }
        ctx.stroke();
    }

    function calculateArea() {
        if (points.length < 3) return;
        
        let area = 0;
        for (let i = 0; i < points.length; i++) {
            let j = (i + 1) % points.length;
            area += points[i].x * points[j].y;
            area -= points[j].x * points[i].y;
        }
        
        // تحويل من بكسل مربع إلى متر مربع
        const finalArea = Math.abs(area / 2) / (pixelsToMeter * pixelsToMeter);
        document.getElementById('areaResult').innerText = `المساحة: ${finalArea.toFixed(2)} م²`;
    }

    function resetPoints() {
        points = [];
        document.getElementById('areaResult').innerText = `المساحة: 0 م²`;
        draw();
    }

    function copyResult() {
        const text = document.getElementById('areaResult').innerText;
        navigator.clipboard.writeText(text);
        alert('تم نسخ النتيجة');
    }
</script>
