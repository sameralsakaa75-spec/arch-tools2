    <div class="controls">
        <label>وقت اليوم (من الشروق للغروب):</label>
        <input type="range" id="timeSlider" min="0" max="180" value="90">
        
        <label>فصل السنة (الميل الشمسي):</label>
        <select id="seasonSelect" style="padding:10px; border-radius:8px;">
            <option value="23.5">الانقلاب الصيفي (أقصى ارتفاع)</option>
            <option value="0" selected>الاعتدال الخريفي/الربيعي</option>
            <option value="-23.5">الانقلاب الشتوي (أدنى ارتفاع)</option>
        </select>

            <span class="label">زاوية الارتفاع (Altitude)</span>
            <span class="value" id="altVal">45°</span>
        </div>
        <div class="data-item">
            <span class="label">طول الظل النسبي</span>
            <span class="value" id="shadowVal">1.0x</span>
        </div>
    </div>

<script>
    const sun = document.getElementById('sun');
    const timeSlider = document.getElementById('timeSlider');
    const seasonSelect = document.getElementById('seasonSelect');
    const altVal = document.getElementById('altVal');
    const shadowVal = document.getElementById('shadowVal');

    function updateSolar() {
        const angle = timeSlider.value; // من 0 إلى 180 درجة
        const seasonOffset = parseFloat(seasonSelect.value);
        
        // حساب الإحداثيات البصرية للقرص الشمسي
        const rad = angle * (Math.PI / 180);
        const x = 90 - (Math.cos(rad) * 45); // تموضع أفقي بنسبة مئوية
        const y = 80 - (Math.sin(rad) * 60 + (seasonOffset * 0.5)); // تموضع رأسي

        sun.style.left = `${x}%`;
        sun.style.top = `${y}%`;

        // حساب زاوية الارتفاع الحقيقية (Altitude)
        let altitude = Math.round(Math.sin(rad) * (90 + seasonOffset));
        if (altitude < 0) altitude = 0;
        altVal.innerText = `${altitude}°`;

        // حساب طول الظل (Cotangent Altitude)
        const shadowFactor = altitude > 0 ? (1 / Math.tan(altitude * Math.PI / 180)).toFixed(2) : "∞";
        shadowVal.innerText = `${shadowFactor}x`;

        // تغيير لون السماء بناءً على الوقت
        const skyColor = `linear-gradient(to bottom, hsl(200, 70%, ${altitude + 20}%), #87ceeb)`;
        document.getElementById('sky').style.background = skyColor;
    }

    timeSlider.addEventListener('input', updateSolar);
    seasonSelect.addEventListener('change', updateSolar);
    updateSolar(); // تشغيل أولي
</script>
