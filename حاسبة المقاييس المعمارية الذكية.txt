<script>
    function calculate() {
        const scale = document.getElementById('scaleSelect').value;
        const drawingVal = document.getElementById('drawingInput').value;
        
        if(drawingVal) {
            // المعادلة: (السنتيمترات * المقياس) / 100 للتحويل لمتر
            const result = (parseFloat(drawingVal) * parseFloat(scale)) / 100;
            document.getElementById('realResult').innerText = result.toLocaleString();
            
            // إضافة اهتزاز بسيط للمس (في حال كان الجهاز يدعم ذلك)
            if (navigator.vibrate) navigator.vibrate(50);
        }
    }

    // الحساب التلقائي أثناء الكتابة
    document.getElementById('drawingInput').addEventListener('input', calculate);
    document.getElementById('scaleSelect').addEventListener('change', calculate);
</script>
