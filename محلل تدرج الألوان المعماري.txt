<script>
    const baseInput = document.getElementById('baseColor');
    
    function generatePalette() {
        const hex = baseInput.value;
        document.getElementById('hexDisplay').innerText = `HEX: ${hex.toUpperCase()}`;
        
        // توليد التدرجات
        renderGrid('shadesGrid', generateShades(hex));
        // توليد اللون المكمل
        renderGrid('compGrid', [hex, getComplementary(hex)]);
        // توليد باليتة معمارية (اللون + رماديات وترابيات)
        renderGrid('moodGrid', [hex, '#f4f1ea', '#d7d1c9', '#5d5c58', '#2d2d2d']);
    }

    function renderGrid(id, colors) {
        const grid = document.getElementById(id);
        grid.innerHTML = '';
        colors.forEach(col => {
            const div = document.createElement('div');
            div.className = 'color-swatch';
            div.style.backgroundColor = col;
            div.innerHTML = `<span class="hex-label">${col.toUpperCase()}</span>`;
            div.onclick = () => {
                navigator.clipboard.writeText(col);
                if (navigator.vibrate) navigator.vibrate(30);
            };
            grid.appendChild(div);
        });
    }

    function generateShades(hex) {
        // تبسيط لتوليد 5 درجات
        return [adjust(hex, 40), adjust(hex, 20), hex, adjust(hex, -20), adjust(hex, -40)];
    }

    function adjust(hex, percent) {
        let num = parseInt(hex.replace("#",""),16),
            amt = Math.round(2.55 * percent),
            R = (num >> 16) + amt, G = (num >> 8 & 0x00FF) + amt, B = (num & 0x0000FF) + amt;
        return "#" + (0x1000000 + (R<255?R<0?0:R:255)*0x10000 + (G<255?G<0?0:G:255)*0x100 + (B<255?B<0?0:B:255)).toString(16).slice(1);
    }

    function getComplementary(hex) {
        let rgb = parseInt(hex.replace("#",""), 16);
        let comp = 0xFFFFFF ^ rgb;
        return "#" + comp.toString(16).padStart(6, '0');
    }

    baseInput.addEventListener('input', generatePalette);
    generatePalette();
</script>
