<script>
    const imageInput = document.getElementById('imageInput');
    const preview = document.getElementById('preview');
    const canvas = document.getElementById('canvas');
    const paletteGrid = document.getElementById('paletteGrid');

    imageInput.addEventListener('change', function(e) {
        const file = e.target.files[0];
        if (!file) return;

        const reader = new FileReader();
        reader.onload = function(event) {
            preview.src = event.target.result;
            preview.style.display = 'block';
            
            const img = new Image();
            img.src = event.target.result;
            img.onload = function() {
                extractColors(img);
            };
        };
        reader.readAsDataURL(file);
    });

    function extractColors(img) {
        const ctx = canvas.getContext('2d');
        canvas.width = img.width;
        canvas.height = img.height;
        ctx.drawImage(img, 0, 0);

        // أخذ عينات من 5 نقاط مختلفة في الصورة
        const points = [
            {x: 0.2, y: 0.2}, {x: 0.5, y: 0.2},
            {x: 0.5, y: 0.5}, {x: 0.2, y: 0.8}, {x: 0.8, y: 0.8}
        ];

        paletteGrid.innerHTML = '';
        points.forEach(p => {
            const pixel = ctx.getImageData(img.width * p.x, img.height * p.y, 1, 1).data;
            const hex = rgbToHex(pixel[0], pixel[1], pixel[2]);
            createSwatch(hex);
        });
    }

    function createSwatch(hex) {
        const div = document.createElement('div');
        div.className = 'swatch';
        div.style.backgroundColor = hex;
        div.innerText = hex.toUpperCase();
        div.onclick = () => {
            navigator.clipboard.writeText(hex);
            if (navigator.vibrate) navigator.vibrate(40);
            alert('تم نسخ اللون: ' + hex);
        };
        paletteGrid.appendChild(div);
    }

    function rgbToHex(r, g, b) {
        return "#" + ((1 << 24) + (r << 16) + (g << 8) + b).toString(16).slice(1);
    }
</script>
